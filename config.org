** EWM daemon fix
EWM creates frames directly via ~make-frame~, not through emacsclient, so
~server-after-make-frame-hook~ never fires. Doom relies on that hook in daemon
mode to apply fonts, theme, and run deferred hooks. Fire it once on the first
graphical frame.

Unlike emacsclient, EWM creates *all* output frames via ~make-frame~, so the
hook must run for every frame (background color, dashboard), not just the first.
Server hooks themselves are guarded by a flag to run exactly once.

The dashboard switch is deferred with ~run-at-time~ because frames are created
before Doom finishes loading (~doom-fallback-buffer~ isn't available yet).

~doom-modeline-update-battery-status~ crashes with ~(args-out-of-range "" 0 -1)~
when battery data is unavailable during daemon init: the icon is not displayable
and the text is empty, so ~(substring "" 0 -1)~ is called. This aborts the
entire ~doom-after-init-hook~ chain, preventing the dashboard from loading.
Advise it to survive the empty-string edge case.

#+begin_src elisp :tangle config.el
(advice-add 'doom-modeline-update-battery-status :around
            (defun +ewm-safe-battery-update-a (fn &rest args)
              "Prevent args-out-of-range on empty battery string."
              (condition-case nil
                  (apply fn args)
                (args-out-of-range nil))))

(when (daemonp)
  ;; `emacs --fg-daemon --eval "(ewm-start-module)"' enters EWM before
  ;; Emacs' daemon command-line code gets a chance to create the normal
  ;; emacsclient socket.  Start it explicitly here so `emacsclient' and the
  ;; `ewm-editor' wrapper can connect to an EWM session.
  (require 'server)
  (when-let* ((name (or (getenv "EMACS_SERVER_NAME")
                        (and (stringp (daemonp)) (daemonp)))))
    (setq server-name name))
  (unless (server-running-p)
    (server-start))

  ;; Track whether server hooks have been triggered (once-only).
  (defvar +ewm--server-hooks-triggered nil)

  (add-hook 'after-make-frame-functions
            (defun +ewm-trigger-server-hooks-h (frame)
              (when (display-graphic-p frame)
                (with-selected-frame frame
                  ;; Run server hooks exactly once (for the first graphic frame).
                  (unless +ewm--server-hooks-triggered
                    (setq +ewm--server-hooks-triggered t)
                    (run-hooks 'server-after-make-frame-hook))
                  ;; Sync frame background from theme and force pgtk to repaint.
                  ;; doom-fix-frame-color-parameters-h runs before the theme loads
                  ;; and only fixes "black" colors, so white persists.
                  (when-let* ((bg (face-background 'default nil t)))
                    (set-frame-parameter frame 'background-color bg))
                  ;; make-frame doesn't consult initial-buffer-choice, so the
                  ;; frame shows *scratch* instead of the Doom dashboard.
                  ;; Defer until after Doom is fully loaded so doom-fallback-buffer
                  ;; and +doom-dashboard-reload are available.
                  (let ((f frame))
                    (run-at-time 0.5 nil
                                 (lambda ()
                                   (when (frame-live-p f)
                                     (with-selected-frame f
                                       (when (and (fboundp 'doom-fallback-buffer)
                                                  (string= (buffer-name) "*scratch*"))
                                         (switch-to-buffer (doom-fallback-buffer))
                                         (when (fboundp '+doom-dashboard-reload)
                                           (+doom-dashboard-reload t)))
                                       ;; Force PGTK to flush the buffer switch to the
                                       ;; Wayland surface.  Without this the frame shows
                                       ;; *scratch* until an input event triggers a repaint.
                                       (redisplay t))))))
                  (redisplay t))))))

** EWM shell editor frames
The shell ~EDITOR~ wrapper in the Nix config uses ~emacsclient~.  For
ordinary server edits, route those client buffers through EWM and mark the
resulting temporary frames with an ~ewm-editor~ frame parameter.  After
~C-x #~ / ~server-edit~ finishes the client edit, close that temporary editor
frame instead of leaving an empty Emacs window in EWM.

Majutsu's ~jjdescription~ support registers its own ~with-editor~ display rule
for jj description files.  Load that support before installing the general EWM
server-window fallback so ~EDITOR=ewm-editor jj describe~ keeps using Majutsu's
editor buffer instead of a temporary EWM frame.

#+begin_src elisp :tangle config.el
(after! server
  (require 'majutsu-jjdescription)

  (defun +ewm-server-window (buffer)
    "Display emacsclient BUFFER in a new EWM frame.

Generic emacsclient editor buffers need an EWM-managed frame with an
`ewm-output' parameter.  Create that frame through EWM, on the currently focused
output, while package-specific `with-editor' display rules (such as Majutsu's jj
description buffer rule) can still override `server-window'."
    (if (and (bound-and-true-p ewm-mode)
             (fboundp 'ewm--focused-frame)
             (fboundp 'ewm--create-frame-for-output))
        (let* ((current-frame (ewm--focused-frame))
               (output (frame-parameter current-frame 'ewm-output))
               (frame (if output
                          (ewm--create-frame-for-output
                           output current-frame
                           '((ewm-editor . t)
                             (ewm-focus-on-create . t)
                             (name . "EWM Editor")))
                        (selected-frame)))
               (window (frame-selected-window frame)))
          (set-window-buffer window buffer)
          ;; A just-created EWM frame does not have its compositor view id yet.
          ;; `select-frame-set-input-focus' is overridden by EWM and tries to
          ;; focus that view id immediately, which fails with
          ;; "Wrong type argument: integerp, nil" for emacsclient editor frames.
          ;; `ewm-focus-on-create' above asks the compositor to focus the frame
          ;; once it is mapped; only force focus when the id already exists.
          (when (frame-parameter frame 'ewm-surface-id)
            (select-frame-set-input-focus frame)
            (select-window window))
          window)
      (pop-to-buffer buffer)))

  (setq server-window #'+ewm-server-window)

  (defun +ewm-delete-editor-frame-after-server-edit-h ()
    "Delete temporary EWM editor frames after `server-edit'."
    (when (and (display-graphic-p)
               (frame-parameter nil 'ewm-editor)
               ;; Do not delete the only remaining frame.
               (cdr (frame-list)))
      (delete-frame nil t)))

  (add-hook 'server-done-hook #'+ewm-delete-editor-frame-after-server-edit-h))
#+end_src

** TODO Shell editors over TRAMP
When a shell is started from an Emacs buffer whose ~default-directory~ is a
TRAMP path, ~with-editor~ can make remote commands use this Emacs as their
editor.  It does not rely on a remote ~emacsclient~ socket; for remote
processes it installs the "sleeping editor" protocol and opens requested files
through TRAMP.

Export the common editor variables so tools with their own priority order, such
as Git and Jujutsu, do not bypass ~with-editor~ via ~VISUAL~, ~GIT_EDITOR~, or
~JJ_EDITOR~.  ~with-editor~ knows about ~shell~, ~eshell~, ~term~, and ~vterm~;
~eat~ and ~ghostel~ need small adapters to inject the same environment before
spawning and to catch the sleeping-editor marker in terminal output.

#+begin_src elisp :tangle no
(defconst +with-editor-editor-envvars '("EDITOR" "VISUAL" "GIT_EDITOR" "JJ_EDITOR"))

(defvar with-editor--envvar)
(defvar with-editor-sleeping-editor-regexp)
(declare-function with-editor--setup "with-editor")
(declare-function with-editor-export-editor "with-editor")
(declare-function with-editor-sleeping-editor-filter "with-editor")

(defun +with-editor--setup-editor-env-vars ()
  "Add editor environment variables for child processes to `process-environment'."
  (require 'with-editor)
  (dolist (envvar +with-editor-editor-envvars)
    (let ((with-editor--envvar envvar))
      (with-editor--setup))))

(defun +with-editor-export-editor-vars-h ()
  "Use this Emacs as editor for commands run in supported shell buffers.

For TRAMP-backed shell buffers, `with-editor' uses its sleeping editor so
remote commands open files through TRAMP in this Emacs instance."
  (require 'with-editor)
  (dolist (envvar +with-editor-editor-envvars)
    (with-editor-export-editor envvar)))

(defun +with-editor--handle-terminal-output (process output)
  "Handle sleeping-editor requests in PROCESS OUTPUT.

Return nil only for chunks that contain a with-editor request and should be
hidden; otherwise return OUTPUT."
  (require 'with-editor)
  (when-let ((buffer (process-buffer process)))
    (with-current-buffer buffer
      (let ((with-editor-sleeping-editor-regexp
             ;; Terminal emulator filters receive raw chunks, so the marker can
             ;; be preceded by prompt text or control sequences.
             (substring with-editor-sleeping-editor-regexp 1)))
        (with-editor-sleeping-editor-filter process output))))
  ;; `with-editor-sleeping-editor-filter' returns nil for incomplete chunks too;
  ;; those must still be rendered.  Only hide actual editor request lines.
  (unless (string-match-p "WITH-EDITOR:" output)
    output))

(dolist (hook '(shell-mode-hook eshell-mode-hook term-exec-hook vterm-mode-hook))
  (add-hook hook #'+with-editor-export-editor-vars-h))

(with-eval-after-load 'eat
  (defun +with-editor-eat-make-a (fn &rest args)
    "Run Eat with this Emacs exported as editor."
    (let ((process-environment process-environment))
      (+with-editor--setup-editor-env-vars)
      (apply fn args)))

  (defun +with-editor-eat-filter-a (fn process output)
    "Let `with-editor' intercept sleeping-editor requests before Eat renders."
    (when-let ((output (+with-editor--handle-terminal-output process output)))
      (funcall fn process output)))

  (unless (advice-member-p #'+with-editor-eat-make-a 'eat-make)
    (advice-add 'eat-make :around #'+with-editor-eat-make-a))
  (unless (advice-member-p #'+with-editor-eat-filter-a 'eat--filter)
    (advice-add 'eat--filter :around #'+with-editor-eat-filter-a)))

(with-eval-after-load 'ghostel
  (defun +with-editor-ghostel-pre-spawn-h ()
    "Run Ghostel with this Emacs exported as editor."
    (+with-editor--setup-editor-env-vars))

  (defun +with-editor-ghostel-filter-a (fn process output)
    "Let `with-editor' intercept sleeping-editor requests before Ghostel renders."
    (when-let ((output (+with-editor--handle-terminal-output process output)))
      (funcall fn process output)))

  (add-hook 'ghostel-pre-spawn-hook #'+with-editor-ghostel-pre-spawn-h)
  (unless (advice-member-p #'+with-editor-ghostel-filter-a 'ghostel--filter)
    (advice-add 'ghostel--filter :around #'+with-editor-ghostel-filter-a)))
#+end_src

(after! org
  (defun get-agenda-category (&optional len)
    "Get category of item at point for agenda.

Category is defined by one of the following items:
- CATEGORY property
- TITLE keyword
- filename without directory and extension

When LEN is a number, resulting string is padded right with
spaces and then truncated with ... on the right if result is
longer than LEN."
    (let* ((file-name (when buffer-file-name
                        (file-name-sans-extension
                         (file-name-nondirectory buffer-file-name))))
           (title (when (derived-mode-p 'org-mode)
                    (org-with-point-at 1
                      (when (re-search-forward (concat "^#\\+title: \\(.*\\)")
                                               (point-max) t)
                        (buffer-substring-no-properties
                         (match-beginning 1)
                         (match-end 1))))))
           (category (if (derived-mode-p 'org-mode)
                         (org-get-category)
                       file-name))
           (result
            (or (if (and title
                         (string-equal category file-name))
                    title
                  category)
                "")))
      (if (numberp len)
          (s-truncate len (s-pad-right len " " result))
        result)))
  
  (setq org-agenda-prefix-format
        '((agenda . "   %i %(get-agenda-category 16)%?-2t %-12s %-6e")
          (todo .   "   %i %(get-agenda-category 16) %-12:t %s")
          (tags .   "   %i %(get-agenda-category 16) %-12:t %s")
          (search . "   %i %(get-agenda-category 16) %s")))

Auto-tag org files with =todo= filetag when they contain TODO entries, then
use those tags to build =org-agenda-files= dynamically.  No org-roam

#+begin_src elisp :tangle config.el
(after! org
  ;; ---- helpers to read/write #+KEYWORD without org-roam ----
  (defun my/org-get-keyword (keyword)
    "Return the value of #+KEYWORD in the current buffer, or nil."
    (save-excursion
      (goto-char (point-min))
      (when (re-search-forward
             (format "^#\\+%s:[ \t]*\\(.*\\)" (regexp-quote keyword))
             nil t)
        (string-trim (match-string-no-properties 1)))))

  (defun my/org-set-keyword (keyword value)
    "Set #+KEYWORD to VALUE in the current buffer.
Insert a new keyword line when none exists yet."
    (save-excursion
      (goto-char (point-min))
      (if (re-search-forward
           (format "^#\\+%s:.*$" (regexp-quote keyword))
           nil t)
          (replace-match (format "#+%s: %s" keyword value))
        ;; Insert after the block of existing keywords at the top
        (goto-char (point-min))
        (while (and (not (eobp)) (looking-at "^[#:]"))
          (forward-line))
        (insert (format "#+%s: %s\n" keyword value)))))

  ;; ---- todo-tag management (org-roam-free) ----
  (defun roam-extra:todo-p ()
    "Return non-nil if current buffer has any TODO entry."
    (when (derived-mode-p 'org-mode)
      (org-element-map (org-element-parse-buffer 'headline) 'headline
        (lambda (h) (eq (org-element-property :todo-type h) 'todo))
        nil 'first-match)))

  (defun roam-extra:update-todo-tag ()
    "Update TODO tag in the current buffer."
    (when (and (not (active-minibuffer-window))
               buffer-file-name
               (file-in-directory-p buffer-file-name org-directory))
      (let* ((tags (roam-extra:get-filetags))
             (is-todo (roam-extra:todo-p)))
        (cond ((and is-todo (not (seq-contains-p tags "todo")))
               (roam-extra:add-filetag "todo"))
              ((and (not is-todo) (seq-contains-p tags "todo"))
               (roam-extra:del-filetag "todo"))))))

  (defun roam-extra:get-filetags ()
    (split-string (or (my/org-get-keyword "filetags") "")))

  (defun roam-extra:add-filetag (tag)
    (let* ((new-tags (cons tag (roam-extra:get-filetags)))
           (new-tags-str (combine-and-quote-strings new-tags)))
      (my/org-set-keyword "filetags" new-tags-str)))

  (defun roam-extra:del-filetag (tag)
    (let* ((new-tags (seq-difference (roam-extra:get-filetags) `(,tag)))
           (new-tags-str (combine-and-quote-strings new-tags)))
      (my/org-set-keyword "filetags" new-tags-str)))

  ;; ---- agenda file collection (no SQLite, Syncthing-safe) ----
  (defun roam-extra:todo-files ()
    "Return a list of org files under `org-directory' whose header
contains a \"todo\" filetag.  Only the first 2 KB of each file is
read, so this stays fast even with thousands of notes."
    (let (files)
      (dolist (f (directory-files-recursively org-directory "\\.org$"))
        (with-temp-buffer
          (insert-file-contents f nil 0 2048)
          (when (re-search-forward "^#\\+filetags:.*\\btodo\\b" nil t)
            (push f files))))
      (nreverse files)))

  (defun roam-extra:update-todo-files (&rest _)
    "Update the value of `org-agenda-files'.
Includes org files with todo filetag and specific additional files."
    (setq org-agenda-files
          (append (roam-extra:todo-files)
                  (mapcar (lambda (file) (expand-file-name file org-directory))
                          default-org-agenda-files))))

  (defvar default-org-agenda-files '("9e_work.org" "cal.org" "aheymans_cal.org" "gedeelde_cal.org"))

  (add-hook 'find-file-hook #'roam-extra:update-todo-tag)
  (add-hook 'before-save-hook #'roam-extra:update-todo-tag)
  (advice-add 'org-agenda :before #'roam-extra:update-todo-files)
  (setq org-agenda-hide-tags-regexp "todo")
  )

#+begin_src elisp :tangle config.el
(require 'json)

(defvar my--openrouter-cache-file
  (expand-file-name "openrouter-models.el" doom-cache-dir)
  "File cache for OpenRouter model list.")

(defun my--openrouter-cache-stale-p ()
  "Return non-nil if the OpenRouter cache is missing or older than 24h."
  (or (not (file-exists-p my--openrouter-cache-file))
      (> (float-time (time-subtract
                      (current-time)
                      (file-attribute-modification-time
                       (file-attributes my--openrouter-cache-file))))
         86400)))

(defun my--openrouter-fetch-models ()
  "Fetch model list from OpenRouter using curl and write to cache file.
Returns the model list, or nil on failure."
  (condition-case err
      (let* ((token (auth-source-pick-first-password :host "openrouter.ai" :user "apikey"))
             (args (append '("--silent" "--max-time" "10"
                             "https://openrouter.ai/api/v1/models")
                           (when token
                             (list "-H" (concat "Authorization: Bearer " token)))))
             (output (with-temp-buffer
                       (when (zerop (apply #'call-process "curl" nil t nil args))
                         (buffer-string)))))
        (when output
          (let* ((json-object-type 'alist)
                 (models (mapcar (lambda (m) (alist-get 'id m))
                                 (alist-get 'data (json-read-from-string output)))))
            (when models
              (with-temp-file my--openrouter-cache-file
                (prin1 models (current-buffer)))
              models))))
    (error
     (message "Warning: Failed to fetch OpenRouter models: %s"
              (error-message-string err))
     nil)))

(defun my--openrouter-models ()
  "Return OpenRouter models from cache, fetching if stale or missing."
  (when (my--openrouter-cache-stale-p)
    (my--openrouter-fetch-models))
  (condition-case nil
      (with-temp-buffer
        (insert-file-contents my--openrouter-cache-file)
        (read (current-buffer)))
    (error '("anthropic/claude-sonnet-4-5-20250929"))))

(defun my/eshell--font-lock-face-to-face ()
  "Copy `font-lock-face' properties to `face' in the current buffer."
  (let ((pos (point-min)) next face)
    (while (< pos (point-max))
      (setq next (next-single-property-change pos 'font-lock-face nil (point-max))
            face (get-text-property pos 'font-lock-face))
      (when face
        (put-text-property pos next 'face face))
      (setq pos next))))

(defun my/eshell-fontify-history-candidate (command)
  "Return COMMAND highlighted like Eshell input in the current directory."
  (let* ((source-dir default-directory)
         ;; `eshell-syntax-highlighting' checks whether commands, directories,
         ;; and file arguments exist.  In a broken TRAMP directory that can block
         ;; history search, so fontify against a local directory instead.
         (fontify-dir (if (file-remote-p source-dir)
                          temporary-file-directory
                        source-dir))
         (key (list fontify-dir command)))
    (or (gethash key my/eshell-history-fontify-cache)
        (puthash key
                 (with-temp-buffer
                   (let ((default-directory fontify-dir)
                         (non-essential t))
                     (require 'eshell-syntax-highlighting)
                     ;; `eshell-syntax-highlighting' works in Eshell buffers and
                     ;; uses Eshell variables/aliases plus `default-directory' to
                     ;; classify commands, options, directories, and files.
                     (delay-mode-hooks (eshell-mode))
                     (let ((inhibit-read-only t))
                       (erase-buffer)
                       (insert command)
                       (goto-char (point-min))
                       (eshell-syntax-highlighting--parse-and-highlight
                        'command (point-max))
                       (my/eshell--font-lock-face-to-face)
                       (buffer-string))))
                 my/eshell-history-fontify-cache))))

(defun my/eshell-consult-history ()
  "Insert an Eshell history entry using Consult with syntax-highlighted candidates."
  (interactive)
  (require 'consult)
  (require 'ring)
  (require 'eshell-atuin)
  (eshell-atuin--history-rotate-cache)
  (eshell-atuin--history-update)
  (let* ((local-history
          (consult--remove-dups (ring-elements eshell-history-ring)))
         (atuin-history
          (delq nil (mapcar #'cdr (eshell-atuin--history-collection))))
         (beg (save-excursion (eshell-bol) (point)))
         (end (pos-eol))
         (candidates
          (consult--remove-dups
           (append (mapcar #'my/eshell-fontify-history-candidate local-history)
                   atuin-history)))
         (selection
          (consult--local-let ((enable-recursive-minibuffers t))
            (consult--read
             (or candidates (user-error "History is empty"))
             :prompt "History: "
             :history t
             :sort nil
             :initial (buffer-substring-no-properties beg end)
             :lookup #'consult--lookup-member
             :state (consult--insertion-preview beg end)))))
    (delete-region beg end)
    (insert (substring-no-properties selection))))

** Tramp
*** Remote shell environments
TRAMP deliberately starts a small POSIX shell with a conservative ~PATH~ for
file operations.  On NixOS this means terminal buffers opened from a remote
~sshx~ directory can otherwise fall back to ~/bin/sh~ and miss the user's Nix
profiles.  Keep TRAMP's file-operation shell alone, but make interactive remote
terminals use the remote login shell and teach TRAMP about user profile paths
when resolving remote programs.

#+begin_src elisp :tangle config.el
(defconst my/tramp-nixos-profile-sentinel-vars
  '("__NIXOS_SET_ENVIRONMENT_DONE" "__ETC_PROFILE_DONE" "__HM_SESS_VARS_SOURCED")
  "Remote NixOS profile guards to unset before starting login shells via TRAMP.")

(defconst my/tramp-nixos-login-shell-command
  (concat "env "
          (mapconcat (lambda (var) (concat "-u " var))
                     my/tramp-nixos-profile-sentinel-vars
                     " ")
          " ${SHELL:-/bin/sh} -l")
  "Shell command that re-enters the remote user's NixOS login shell.")

(after! tramp
  (require 'seq)
  ;; Include paths that are normally added by NixOS login/profile setup when
  ;; TRAMP searches for remote programs.  `tramp-own-remote-path' asks TRAMP to
  ;; read the user's remote profile path; the explicit Nix paths cover common
  ;; NixOS profile locations even when that detection is incomplete.
  (setq tramp-remote-path
        (append (list 'tramp-own-remote-path
                      "~/.nix-profile/bin"
                      "~/.local/state/nix/profile/bin"
                      "/nix/profile/bin"
                      "/nix/var/nix/profiles/default/bin"
                      "/run/wrappers/bin"
                      "/run/current-system/sw/bin")
                (seq-difference tramp-remote-path
                                '(tramp-own-remote-path
                                  "~/.nix-profile/bin"
                                  "~/.local/state/nix/profile/bin"
                                  "/nix/profile/bin"
                                  "/nix/var/nix/profiles/default/bin"
                                  "/run/wrappers/bin"
                                  "/run/current-system/sw/bin")
                                #'equal))))

(after! vterm
  ;; vterm's `login-shell' sentinel selects the user's shell path but does not
  ;; invoke it as a login shell.  Use a shell command so NixOS profiles are
  ;; sourced by bash/zsh/fish when terminals are opened from TRAMP buffers.
  (setq vterm-tramp-shells
        `(("ssh" ,my/tramp-nixos-login-shell-command)
          ("sshx" ,my/tramp-nixos-login-shell-command)
          ("scp" ,my/tramp-nixos-login-shell-command)
          ("rpc" ,my/tramp-nixos-login-shell-command)
          (t ,my/tramp-nixos-login-shell-command)
          ("docker" "/bin/sh"))))

(after! eat
  ;; Eat expects shell commands rather than a `login-shell' sentinel.
  (setq eat-tramp-shells
        `(("ssh" . ,(concat "exec " my/tramp-nixos-login-shell-command))
          ("sshx" . ,(concat "exec " my/tramp-nixos-login-shell-command))
          ("scp" . ,(concat "exec " my/tramp-nixos-login-shell-command))
          ("rpc" . ,(concat "exec " my/tramp-nixos-login-shell-command))
          ("docker" . "/bin/sh"))))

(after! ghostel
  (defun my/ghostel-unset-nixos-profile-sentinels-h ()
    "Let remote NixOS login shells rebuild their profile environment."
    (when (file-remote-p default-directory)
      (dolist (var my/tramp-nixos-profile-sentinel-vars)
        (setenv var nil))))

  (add-hook 'ghostel-pre-spawn-hook #'my/ghostel-unset-nixos-profile-sentinels-h)

  ;; Ghostel now detects remote login shells itself and supplies -l/-i.
  (setq ghostel-tramp-shells
        '(("docker" "/bin/sh")
          ("rpc" login-shell)

* Temporary fixes
** ibuffer deleting the final line
Emacs 30.2's ~ibuffer-map-lines~ deletes lines with
~(1+ (line-end-position))~.  When the final Ibuffer entry is selected and the
buffer has no trailing newline or summary line, that end position is past
~point-max~, so ~ibuffer-do-delete~ raises ~args-out-of-range~.

Use the beginning of the next line instead; it is the same position when a
newline is present and safely clamps to ~point-max~ on the final unterminated
line.

Also normalize Doom's current +icons Ibuffer format.  It contains
~(vc-status 12 :left)~, which Emacs 30 reads as MIN = 12 and MAX = ~:left~;
when the VC status is wide enough, redisplay fails with
~wrong-type-argument number-or-marker-p, :left~.  The normalized form is
~(vc-status 12 12 :left)~.

#+begin_src elisp :tangle config.el
(after! ibuffer
  (defun my/ibuffer-normalize-format-entry (entry)
    "Return ENTRY with legacy `(COLUMN MIN ALIGN ...)' syntax normalized.
Emacs 30 expects `(COLUMN MIN MAX ALIGN ...)' when an Ibuffer column
specifies alignment.  Doom's +icons Ibuffer format currently contains
`(vc-status 12 :left)', which makes `:left' the MAX width and eventually
raises `wrong-type-argument number-or-marker-p'."
    (if (and (consp entry)
             (integerp (nth 1 entry))
             (keywordp (nth 2 entry)))
        (append (list (nth 0 entry) (nth 1 entry) (nth 1 entry))
                (nthcdr 2 entry))
      entry))

  (setq ibuffer-formats
        (mapcar (lambda (format)
                  (mapcar #'my/ibuffer-normalize-format-entry format))
                ibuffer-formats)
        ibuffer-compiled-formats nil)
  (when (boundp 'ibuffer-compiled-filter-formats)
    (setq ibuffer-compiled-filter-formats nil))

  (defun ibuffer-map-lines (function &optional nomodify group)
    "Call FUNCTION for each buffer.
Set the ibuffer modification flag unless NOMODIFY is non-nil.

If optional argument GROUP is non-nil, then only call FUNCTION on
buffers in filtering group GROUP.

FUNCTION is called with two arguments:
the buffer object itself and the current mark symbol."
    (ibuffer-assert-ibuffer-mode)
    (ibuffer-forward-line 0)
    (let* ((orig-target-line (1+ (count-lines (save-excursion
                                                (goto-char (point-min))
                                                (ibuffer-forward-line 0)
                                                (point))
                                              (point))))
           (target-line-offset orig-target-line)
           (ibuffer-map-lines-total 0)
           (ibuffer-map-lines-count 0))
      (unwind-protect
          (progn
            (setq buffer-read-only nil)
            (goto-char (point-min))
            (ibuffer-forward-line 0 t)
            (while (and (not (eobp))
                        (not (get-text-property (point) 'ibuffer-summary))
                        (progn
                          (ibuffer-forward-line 0 t)
                          (and (not (eobp))
                               (not (get-text-property (point) 'ibuffer-summary)))))
              (let ((result
                     (if (buffer-live-p (ibuffer-current-buffer))
                         (when (or (null group)
                                   (when-let ((it (get-text-property
                                                   (point) 'ibuffer-filter-group)))
                                     (equal group it)))
                           (save-excursion
                             (funcall function
                                      (ibuffer-current-buffer)
                                      (ibuffer-current-mark))))
                       ;; Kill the line if the buffer is dead.
                       'kill)))
                ;; A given mapping function should return:
                ;; nil if it chose not to affect the buffer
                ;; `kill' means remove the line from the buffer list
                ;; t otherwise
                (cl-incf ibuffer-map-lines-total)
                (cond ((null result)
                       (forward-line 1))
                      ((eq result 'kill)
                       ;; Emacs 30.2 used (1+ (line-end-position)) here, which
                       ;; is out of range on an unterminated final line.
                       (delete-region (line-beginning-position)
                                      (line-beginning-position 2))
                       (cl-incf ibuffer-map-lines-count)
                       (when (< ibuffer-map-lines-total
                                orig-target-line)
                         (cl-decf target-line-offset)))
                      (t
                       (cl-incf ibuffer-map-lines-count)
                       (forward-line 1)))))
            ;; With `ibuffer-auto-mode' enabled, `ibuffer-expert' nil
            ;; and more than one marked buffer lines, the preceding loop
            ;; counts the automatically popped up (and hence not
            ;; user-marked) buffer "*Ibuffer confirmation*".  Since
            ;; Ibuffer reports how many marked buffers lines were acted
            ;; upon, and in this case the reported count would be too
            ;; high by one, we decrement the count to avoid the
            ;; confusing message (see bug#64230).
            (if (and (featurep 'ibuf-ext) ibuffer-auto-mode
                     (> ibuffer-map-lines-count 1)
                     (not ibuffer-expert))
                (1- ibuffer-map-lines-count)
              ibuffer-map-lines-count))
        (progn
          (setq buffer-read-only t)
          (unless nomodify
            (set-buffer-modified-p nil))
          (goto-char (point-min))
          (ibuffer-forward-line 0)
          (ibuffer-forward-line (1- target-line-offset)))))))
#+end_src
